📊 Variables & Data Types

Learn to store and work with different types of data in JavaScript

📦 What are Variables?

Variables là containers để store data values trong JavaScript. Think of them như labeled boxes nơi bạn có thể put data vào và retrieve nó later.

// Variable declarations in JavaScript
let name = "John";          // String variable
let age = 25;              // Number variable
let isStudent = true;      // Boolean variable

// Using variables
console.log(name);         // Output: John
console.log(age);          // Output: 25
console.log(isStudent);   // Output: true

🔄 Variable Declaration Keywords

Keyword Scope Reassignable Hoisting When to Use
let Block scope ✅ Yes ❌ No Default choice for variables
const Block scope ❌ No ❌ No Constants, objects, arrays
var Function scope ✅ Yes ✅ Yes Legacy code (avoid in modern JS)

🎯 JavaScript Data Types

JavaScript có dynamic typing - biến có thể hold different types of data without declaring the type explicitly.

🔢 Primitive Data Types

String

Text data in quotes

"Hello World"
'JavaScript'
`Template string`

Number

Integers and decimals

42
3.14
-17
Infinity

Boolean

True or false values

true
false

Undefined

Variable declared but not assigned

let x;
console.log(x);
// undefined

Null

Intentionally empty value

let data = null;
// explicitly empty

Symbol

Unique identifier

Symbol('id')
Symbol.for('key')

🏗️ Non-Primitive (Reference) Types

// Object
let person = {
    name: "Alice",
    age: 30,
    isEmployed: true
};

// Array
let colors = ["red", "green", "blue"];

// Function
function greet() {
    return "Hello!";
}

🔍 Type Checking

Use typeof operator để check the type of a variable:

// typeof operator examples
console.log(typeof "Hello");        // "string"
console.log(typeof 42);           // "number"
console.log(typeof true);         // "boolean"
console.log(typeof undefined);   // "undefined"
console.log(typeof null);        // "object" (JavaScript quirk!)
console.log(typeof {});           // "object"
console.log(typeof []);           // "object"
console.log(typeof function(){});  // "function"

🔤 Working with Strings

String Creation

// Different ways to create strings
let str1 = "Double quotes";
let str2 = 'Single quotes';
let str3 = `Template literals`;

// Template literals với variables
let name = "Alice";
let age = 25;
let message = `Hello, my name is ${name} and I'm ${age} years old.`;

console.log(message);

String Methods

let text = "JavaScript Programming";

// Common string methods
console.log(text.length);          // 21
console.log(text.toUpperCase());   // "JAVASCRIPT PROGRAMMING"
console.log(text.toLowerCase());   // "javascript programming"
console.log(text.charAt(0));       // "J"
console.log(text.indexOf("Script")); // 4
console.log(text.slice(0, 10));    // "JavaScript"

🔢 Working with Numbers

// Number operations
let x = 10;
let y = 3;

console.log(x + y);    // Addition: 13
console.log(x - y);    // Subtraction: 7
console.log(x * y);    // Multiplication: 30
console.log(x / y);    // Division: 3.333...
console.log(x % y);    // Modulus: 1
console.log(x ** y);   // Exponentiation: 1000

// Math object methods
console.log(Math.round(4.7));    // 5
console.log(Math.ceil(4.1));     // 5
console.log(Math.floor(4.9));    // 4
console.log(Math.random());    // Random between 0-1

🔄 Type Conversion

Implicit Conversion (Coercion)

// JavaScript automatically converts types
console.log("5" + 3);      // "53" (string concatenation)
console.log("5" - 3);      // 2 (numeric subtraction)
console.log("5" * 3);      // 15 (numeric multiplication)
console.log(true + 1);     // 2 (true becomes 1)
console.log(false + 1);    // 1 (false becomes 0)

Explicit Conversion

// Converting to string
let num = 123;
console.log(String(num));        // "123"
console.log(num.toString());     // "123"

// Converting to number
let str = "456";
console.log(Number(str));        // 456
console.log(parseInt(str));     // 456
console.log(parseFloat("3.14")); // 3.14

// Converting to boolean
console.log(Boolean(1));         // true
console.log(Boolean(0));         // false
console.log(Boolean(""));        // false
console.log(Boolean("hello"));   // true
