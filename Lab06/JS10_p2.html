üì§ Spread & Rest Operators (...)

The spread operator (...) expands elements, while rest operator (...) collects elements.

Spread Operator

// Array spreading
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const combined = [...arr1, ...arr2]; // [1, 2, 3, 4, 5, 6]

// Object spreading
const person = {name: "Alice", age: 25};
const contact = {email: "alice@email.com", phone: "123-456"};
const fullProfile = {...person, ...contact};

// Function arguments
const numbers = [1, 2, 3, 4, 5];
console.log(Math.max(...numbers)); // 5

// Copying arrays/objects
const originalArray = [1, 2, 3];
const copiedArray = [...originalArray];

const originalObj = {a: 1, b: 2};
const copiedObj = {...originalObj};

Rest Parameters

// Rest in function parameters
function sum(...numbers) {
    return numbers.reduce((acc, num) => acc + num, 0);
}

console.log(sum(1, 2, 3, 4, 5)); // 15

// Rest with destructuring
const [first, second, ...rest] = [1, 2, 3, 4, 5];
// first = 1, second = 2, rest = [3, 4, 5]

const {name, age, ...otherProps} = {
    name: "Bob",
    age: 30,
    city: "Hanoi",
    country: "Vietnam"
};

// Mixed parameters
function logMessage(level, message, ...details) {
    console.log(`[${level}] ${message}`);
    if (details.length > 0) {
        console.log("Details:", details);
    }
}

‚öôÔ∏è Default Parameters

Default parameters allow function parameters to have default values when no argument is passed.

// Basic default parameters
function greet(name = "Guest", greeting = "Hello") {
    return `${greeting}, ${name}!`;
}

console.log(greet()); // "Hello, Guest!"
console.log(greet("Alice")); // "Hello, Alice!"
console.log(greet("Bob", "Hi")); // "Hi, Bob!"

// Complex default values
function createUser(
    name = "Anonymous",
    id = Date.now(),
    permissions = ["read"],
    settings = {theme: "light", lang: "en"}
) {
    return {name, id, permissions, settings};
}

// Default parameters with destructuring
function processOrder({
    items = [],
    discount = 0,
    shipping = 10,
    currency = "USD"
} = {}) {
    const subtotal = items.reduce((sum, item) => sum + item.price, 0);
    const total = subtotal - discount + shipping;
    return `Total: ${total} ${currency}`;
}

üèóÔ∏è ES6 Classes

ES6 Classes provide a cleaner syntax for creating objects and handling inheritance.

Basic Class Syntax

// Class declaration
class Person {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }
    
    // Method
    greet() {
        return `Hi, I'm ${this.name} and I'm ${this.age} years old`;
    }
    
    // Static method
    static species() {
        return "Homo sapiens";
    }
    
    // Getter
    get info() {
        return `${this.name} (${this.age})`;
    }
    
    // Setter
    set fullName(value) {
        [this.firstName, this.lastName] = value.split(" ");
    }
}

// Usage
const person = new Person("Alice", 25);
console.log(person.greet());
console.log(Person.species());

Class Inheritance

// Inheritance with extends
class Student extends Person {
    constructor(name, age, studentId, major) {
        super(name, age); // Call parent constructor
        this.studentId = studentId;
        this.major = major;
    }
    
    // Override parent method
    greet() {
        return `${super.greet()}. I study ${this.major}`;
    }
    
    // New method
    study() {
        return `${this.name} is studying ${this.major}`;
    }
}

const student = new Student("Bob", 20, "S123", "Computer Science");
console.log(student.greet());
console.log(student.study());

üì¶ ES6 Modules

ES6 Modules provide a way to organize and share code between files using import v√† export.

Export Syntax

// math.js - Named exports
export const PI = 3.14159;

export function add(a, b) {
    return a + b;
}

export function multiply(a, b) {
    return a * b;
}

// Alternative export syntax
const subtract = (a, b) => a - b;
const divide = (a, b) => a / b;

export {subtract, divide};

// Default export
export default class Calculator {
    constructor() {
        this.result = 0;
    }
    
    add(n) {
        this.result += n;
        return this;
    }
    
    getResult() {
        return this.result;
    }
}

Import Syntax

// app.js - Import examples

// Named imports
import {add, multiply, PI} from './math.js';

// Import all as namespace
import * as MathUtils from './math.js';

// Default import
import Calculator from './math.js';

// Mixed imports
import Calculator, {add, PI} from './math.js';

// Import with alias
import {add as sum, multiply as mult} from './math.js';

// Dynamic imports
async function loadMath() {
    const mathModule = await import('./math.js');
    console.log(mathModule.add(2, 3));
}
