ü§î Conditional Statements

Conditional statements allow your code ƒë·ªÉ make decisions based on different conditions. They control which blocks of code execute.

if Statement

// Basic if statement
let age = 18;

if (age >= 18) {
    console.log("You can vote!");
}

// if...else statement
if (age >= 18) {
    console.log("You are an adult");
} else {
    console.log("You are a minor");
}

// if...else if...else statement
let score = 85;

if (score >= 90) {
    console.log("Grade A");
} else if (score >= 80) {
    console.log("Grade B");
} else if (score >= 70) {
    console.log("Grade C");
} else {
    console.log("Grade F");
}

switch Statement

// switch statement for multiple conditions
let day = "Monday";

switch (day) {
    case "Monday":
        console.log("Start of work week");
        break;
    case "Wednesday":
        console.log("Hump day!");
        break;
    case "Friday":
        console.log("TGIF!");
        break;
    case "Saturday":
    case "Sunday":
        console.log("Weekend!");
        break;
    default:
        console.log("Regular day");
}

üîÅ Loops

Loops allow you ƒë·ªÉ repeat code multiple times. JavaScript has several types of loops cho different scenarios.

for Loop

// Basic for loop
for (let i = 0; i < 5; i++) {
    console.log(`Count: ${i}`);
}

// Counting backwards
for (let i = 5; i > 0; i--) {
    console.log(`Countdown: ${i}`);
}

// Loop through an array
let fruits = ["apple", "banana", "orange"];
for (let i = 0; i < fruits.length; i++) {
    console.log(`Fruit ${i + 1}: ${fruits[i]}`);
}

while Loop

// while loop - runs while condition is true
let count = 0;
while (count < 3) {
    console.log(`While count: ${count}`);
    count++;
}

// do...while loop - runs at least once
let num = 0;
do {
    console.log(`Do-while: ${num}`);
    num++;
} while (num < 3);

for...of v√† for...in Loops

// for...of loop - iterates over values
let colors = ["red", "green", "blue"];
for (let color of colors) {
    console.log(`Color: ${color}`);
}

// for...in loop - iterates over keys/indices
let person = {name: "Alice", age: 25, city: "New York"};
for (let key in person) {
    console.log(`${key}: ${person[key]}`);
}

üõë Loop Control

break v√† continue

// break - exits the loop completely
for (let i = 0; i < 10; i++) {
    if (i === 5) {
        break; // Stop loop when i is 5
    }
    console.log(i); // Prints 0, 1, 2, 3, 4
}

// continue - skips current iteration
for (let i = 0; i < 5; i++) {
    if (i === 2) {
        continue; // Skip when i is 2
    }
    console.log(i); // Prints 0, 1, 3, 4
}

üîç Finding Elements

let numbers = [1, 3, 7, 9, 12];
let target = 7;
let found = false;

for (let i = 0; i < numbers.length; i++) {
    if (numbers[i] === target) {
        found = true;
        break;
    }
}

üìä Counting

let text = "hello world";
let vowelCount = 0;

for (let char of text) {
    if ("aeiou".includes(char)) {
        vowelCount++;
    }
}

üìà Accumulating

let numbers = [1, 2, 3, 4, 5];
let sum = 0;

for (let num of numbers) {
    sum += num;
}
// sum = 15

üîÑ Building Arrays

let squares = [];

for (let i = 1; i <= 5; i++) {
    squares.push(i * i);
}
// squares = [1, 4, 9, 16, 25]
