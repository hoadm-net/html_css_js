üéØ Selecting Elements

Before you can manipulate elements, you need to select them from the DOM:

Basic Selection Methods

// Select by ID
let element = document.getElementById("myId");

// Select by class name (returns HTMLCollection)
let elements = document.getElementsByClassName("myClass");

// Select by tag name
let paragraphs = document.getElementsByTagName("p");

// Modern CSS selectors (recommended)
let singleElement = document.querySelector("#myId"); // First match
let multipleElements = document.querySelectorAll(".myClass"); // All matches

// Complex selectors
let specificElement = document.querySelector("div.container > p:first-child");
let allButtons = document.querySelectorAll("button[type='submit']");

Selection Examples

// Check if element exists before using
let myElement = document.getElementById("header");
if (myElement) {
    console.log("Element found!");
} else {
    console.log("Element not found!");
}

// Working with NodeList (querySelectorAll)
let allParagraphs = document.querySelectorAll("p");
allParagraphs.forEach((paragraph, index) => {
    console.log(`Paragraph ${index}: ${paragraph.textContent}`);
});

// Convert HTMLCollection to Array
let divs = document.getElementsByTagName("div");
let divsArray = Array.from(divs);
divsArray.forEach(div => {
    console.log(div.className);
});

‚úèÔ∏è Manipulating Elements

Changing Content

// Change text content
let heading = document.querySelector("h1");
heading.textContent = "New Heading Text";

// Change HTML content
let container = document.getElementById("container");
container.innerHTML = "<p>New <strong>HTML</strong> content</p>";

// Safer alternative to innerHTML
container.textContent = "Safe text content"; // Won't execute scripts

// Get current content
let currentText = heading.textContent;
let currentHTML = container.innerHTML;
console.log("Current text:", currentText);
console.log("Current HTML:", currentHTML);

Changing Attributes

// Set attributes
let image = document.querySelector("img");
image.setAttribute("src", "new-image.jpg");
image.setAttribute("alt", "New description");

// Direct property access (preferred for common attributes)
image.src = "another-image.jpg";
image.alt = "Another description";

// Get attributes
let imageSrc = image.getAttribute("src");
let imageAlt = image.alt; // Direct property

// Check if attribute exists
if (image.hasAttribute("data-id")) {
    console.log("Has data-id attribute");
}

// Remove attributes
image.removeAttribute("data-old");

Changing Styles

// Direct style manipulation
let element = document.getElementById("myElement");
element.style.color = "red";
element.style.backgroundColor = "yellow";
element.style.fontSize = "20px";
element.style.display = "none"; // Hide element

// CSS classes (recommended approach)
element.className = "new-class"; // Replace all classes
element.className += " additional-class"; // Add class

// Modern classList methods
element.classList.add("highlight"); // Add class
element.classList.remove("old-class"); // Remove class
element.classList.toggle("active"); // Toggle class
element.classList.contains("highlight"); // Check if has class

// Multiple style changes
Object.assign(element.style, {
    color: "blue",
    backgroundColor: "lightgray",
    padding: "10px",
    border: "2px solid black"
});

‚ûï Creating & Adding Elements

// Create new elements
let newParagraph = document.createElement("p");
newParagraph.textContent = "This is a new paragraph";
newParagraph.className = "new-paragraph";

// Create text nodes
let textNode = document.createTextNode("Just some text");

// Add elements to the DOM
let container = document.getElementById("container");

// Append to end
container.appendChild(newParagraph);

// Insert at specific position
let firstChild = container.firstElementChild;
container.insertBefore(newParagraph, firstChild);

// Modern methods (more flexible)
container.prepend(newParagraph); // Add to beginning
container.append(newParagraph); // Add to end
firstChild.before(newParagraph); // Insert before element
firstChild.after(newParagraph); // Insert after element

// Create complex structures
let newDiv = document.createElement("div");
newDiv.innerHTML = `
    <h3>New Section</h3>
    <p>Content goes here</p>
    <button>Click me</button>
`;
container.append(newDiv);

Removing Elements

// Remove element (modern way)
let elementToRemove = document.querySelector(".remove-me");
if (elementToRemove) {
    elementToRemove.remove();
}

// Remove element (traditional way)
let parent = elementToRemove.parentNode;
if (parent) {
    parent.removeChild(elementToRemove);
}

// Remove all children
let container = document.getElementById("container");
container.innerHTML = ""; // Quick but not always best

// Remove children one by one (safer)
while (container.firstChild) {
    container.removeChild(container.firstChild);
}

// Modern alternative
container.replaceChildren(); // Remove all children

üß≠ DOM Navigation

// Parent navigation
let element = document.querySelector(".child");
let parent = element.parentNode; // Any parent node
let parentElement = element.parentElement; // Parent element only

// Children navigation
let container = document.getElementById("container");
let allChildren = container.children; // HTMLCollection of elements
let allChildNodes = container.childNodes; // NodeList (includes text nodes)
let firstChild = container.firstElementChild;
let lastChild = container.lastElementChild;

// Sibling navigation
let nextSibling = element.nextElementSibling;
let previousSibling = element.previousElementSibling;

// Find specific elements
let closestSection = element.closest("section"); // Find nearest ancestor
let matchingChildren = container.querySelectorAll(":scope > .direct-child");
