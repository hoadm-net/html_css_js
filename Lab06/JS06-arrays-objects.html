üìä Arrays - Ordered Collections

Arrays are ordered lists of items. They can store multiple values in a single variable v√† provide methods for manipulation.

üóÇÔ∏è Array Structure

Arrays use zero-based indexing:

Index: 0
Value: "apple"
Index: 1
Value: "banana"
Index: 2
Value: "orange"
Index: 3
Value: "grape"

Creating Arrays

// Array literal (most common)
let fruits = ["apple", "banana", "orange"];

// Array constructor
let numbers = new Array(1, 2, 3, 4, 5);

// Empty array
let empty = [];

// Mixed data types
let mixed = ["text", 42, true, null, {name: "object"}];

// Accessing elements
console.log(fruits[0]); // "apple"
console.log(fruits[1]); // "banana"
console.log(fruits.length); // 3

Array Methods - Adding/Removing

let animals = ["cat", "dog"];

// Add to end
animals.push("bird"); // ["cat", "dog", "bird"]

// Add to beginning
animals.unshift("fish"); // ["fish", "cat", "dog", "bird"]

// Remove from end
let lastAnimal = animals.pop(); // "bird", array: ["fish", "cat", "dog"]

// Remove from beginning
let firstAnimal = animals.shift(); // "fish", array: ["cat", "dog"]

// Remove/add at specific position
animals.splice(1, 0, "rabbit"); // ["cat", "rabbit", "dog"]
animals.splice(1, 1); // Remove "rabbit": ["cat", "dog"]

Array Methods - Searching/Checking

let colors = ["red", "green", "blue", "yellow"];

// Find index
console.log(colors.indexOf("blue")); // 2
console.log(colors.indexOf("purple")); // -1 (not found)

// Check if includes
console.log(colors.includes("green")); // true
console.log(colors.includes("orange")); // false

// Find with condition
let numbers = [1, 5, 10, 15, 20];
let firstLarge = numbers.find(num => num > 10); // 15
let largeIndex = numbers.findIndex(num => num > 10); // 3

Array Methods - Transformation

let numbers = [1, 2, 3, 4, 5];

// Map - transform each element
let doubled = numbers.map(num => num * 2);
console.log(doubled); // [2, 4, 6, 8, 10]

// Filter - select elements
let evenNumbers = numbers.filter(num => num % 2 === 0);
console.log(evenNumbers); // [2, 4]

// Reduce - combine into single value
let sum = numbers.reduce((acc, num) => acc + num, 0);
console.log(sum); // 15

// Sort and reverse
let words = ["banana", "apple", "cherry"];
console.log(words.sort()); // ["apple", "banana", "cherry"]
console.log(words.reverse()); // ["cherry", "banana", "apple"]

üèõÔ∏è Objects - Key-Value Collections

Objects store data as key-value pairs, allowing you to organize related information together.

üóÉÔ∏è Object Structure

Objects use keys to access values:

Key: "name"
Value: "John"
Key: "age"
Value: 25
Key: "city"
Value: "Hanoi"
Key: "isStudent"
Value: true

Creating Objects

// Object literal (most common)
let person = {
    name: "Alice",
    age: 25,
    city: "Hanoi",
    isStudent: true
};

// Object constructor
let car = new Object();
car.brand = "Toyota";
car.model = "Camry";
car.year = 2022;

// Accessing properties
console.log(person.name); // "Alice"
console.log(person["age"]); // 25

// Adding/modifying properties
person.email = "alice@email.com";
person.age = 26;

Object Methods

let student = {
    name: "Bob",
    age: 20,
    subjects: ["Math", "Physics", "Chemistry"],
    
    // Method inside object
    introduce: function() {
        return `Hi, I'm ${this.name} and I'm ${this.age} years old.`;
    },
    
    // Arrow function method (ES6)
    getSubjects: () => {
        return student.subjects.join(", ");
    }
};

console.log(student.introduce()); // "Hi, I'm Bob and I'm 20 years old."
console.log(student.getSubjects()); // "Math, Physics, Chemistry"

Object Utility Methods

let product = {
    id: 1,
    name: "Laptop",
    price: 999,
    category: "Electronics"
};

// Get all keys
let keys = Object.keys(product);
console.log(keys); // ["id", "name", "price", "category"]

// Get all values
let values = Object.values(product);
console.log(values); // [1, "Laptop", 999, "Electronics"]

// Get key-value pairs
let entries = Object.entries(product);
console.log(entries); // [["id", 1], ["name", "Laptop"], ...]

// Check if property exists
console.log("name" in product); // true
console.log(product.hasOwnProperty("price")); // true

// Delete property
delete product.category;
console.log(product); // {id: 1, name: "Laptop", price: 999}

üöÄ Advanced Array & Object Techniques

Destructuring Assignment

// Array destructuring
let colors = ["red", "green", "blue"];
let [primary, secondary, tertiary] = colors;
console.log(primary); // "red"

// Object destructuring
let person = {name: "Alice", age: 25, city: "Hanoi"};
let {name, age} = person;
console.log(name); // "Alice"

// Nested destructuring
let data = {
    user: {
        profile: {
            name: "Bob",
            email: "bob@email.com"
        }
    }
};
let {user: {profile: {name: userName}}} = data;
console.log(userName); // "Bob"

Spread Operator

// Array spreading
let arr1 = [1, 2, 3];
let arr2 = [4, 5, 6];
let combined = [...arr1, ...arr2]; // [1, 2, 3, 4, 5, 6]

// Object spreading
let obj1 = {a: 1, b: 2};
let obj2 = {c: 3, d: 4};
let merged = {...obj1, ...obj2}; // {a: 1, b: 2, c: 3, d: 4}

// Copy arrays/objects
let originalArray = [1, 2, 3];
let copiedArray = [...originalArray];

let originalObject = {name: "Alice", age: 25};
let copiedObject = {...originalObject};

JSON - Data Exchange

// Object to JSON string
let user = {
    name: "Alice",
    age: 25,
    hobbies: ["reading", "coding"],
    address: {
        city: "Hanoi",
        country: "Vietnam"
    }
};

let jsonString = JSON.stringify(user);
console.log(jsonString);

// JSON string to object
let parsedUser = JSON.parse(jsonString);
console.log(parsedUser.name); // "Alice"

// Pretty print JSON
let prettyJson = JSON.stringify(user, null, 2);
console.log(prettyJson);
